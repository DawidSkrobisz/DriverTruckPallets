create database dritrucpal;
CREATE TABLE Driver (
                        driver_id INT PRIMARY KEY,
                        first_name VARCHAR(255),
                        last_name VARCHAR(255),
                        psych_test_date DATE,
                        medical_test_date DATE,
                        license_expiry_date DATE
);

CREATE TABLE Truck (
                       truck_id INT PRIMARY KEY,
                       model VARCHAR(255),
                       license_plate VARCHAR(20),
                       mileage INT,
                       vin_number VARCHAR(50),
                       inspection_date DATE,
                       insurance_date DATE
);

CREATE TABLE Company (
                         company_id INT PRIMARY KEY,
                         positive_palettes INT,
                         negative_palettes INT,
                         palette_balance INT,
                         loading_date DATE
);

CREATE TABLE Palette (
                         palette_id INT PRIMARY KEY,
                         loaded_palettes INT,
                         exchanged_palettes INT,
                         returned_palettes INT
);
